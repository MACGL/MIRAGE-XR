2 /* ThreeLSControl
3  *
4  * Author: Gerard Llorach
5  * Date: 20/03/2018
6  * Version: 0.0
7  *
8  * DESCRIPTION:
9  * This script connects the ThreeLS class with the scene. ThreeLS needs an
10  * AudioSource and returns weights from 0 and 1 for the blend shapes Kiss,
11  * Lips Closed and Mouth Open (in FACS its AU22, AU24 and AU27). This script
12  * finds the blend shapes of the SkinnedMeshRenderers that match the input name
13  * and assigns the weights from ThreeLS to the SkinnedMeshRenderers.
14  *
15  * USE:
16  * 0. Watch the video explaining how to use it (found in the description of the asset)
17  * 1. Add this script to the virtual character. The objects with the blend shapes
18  * to modify should be children of this object.
19  * 2. Connect your Audio Source to this script (Audio Input variable)
20  * 3. In the editor, write down the name of the blend shapes to modify under
21  * each corresponding input string array (Kiss Blend Shape Names, Lips Closed
22  * Blend Shape Names, Mouth Open Blend Shape Names). You will find the name of
23  * the blend shapes in the Skinned Mesh Renderer component of the head/mouth/teeth
24  * of your virtual character.
25  * 4. Play the scene. If the Audio Source is playing, you should see the lips
26  * of your virtual character moving. You will get error messages if the blend
27  * shape names you wrote down are not found in any of the child meshes.
28  * 5. Play around with the sliders Threshold, Smoothness and Vocal Track Factor
29  * as well as the factors for each blend shape.
30  * Threshold helps when the audio is a bit noise. The smaller the value,
31  * less sentitiveness. Smoothness makes the transitions of the lips more smooth.
32  * Vocal Track Factor tries to take into account differences between speakers.
33  * Values around 0.7-0.8 have been found to be good for female voices.
34  *
35  * OTHER COMMENTS:
36  * This lipsync strategy should work also with a microphone, although I haven't
37  * seen a straight forward way to connect the microphone to the spectrum analysis.
38  * Remains to be done I guess.
39  *
40  */

180  // Lipsync values
181  private void updateBlendShapes()
182  {
183  // Blend shapes fade to zero when audio stops
184  if (!audioInput.isPlaying && wasPlaying)
185  {// Start decay of lipsync values when audio stopped
186  wasPlaying = false;
187  fadingOut = true;
188  timeToFadeOut = Time.time + 1.0f; // One second fading out
189  }
190  if (!audioInput.isPlaying && fadingOut)
191  { // Fade out lip values to zero
192  fadeOutLips();
193  return;
194  }
195  else if (!audioInput.isPlaying)
196  return;
197 
198  // Assign weights (kiss, lipsClosed, mouthOpen)
199  for (int i = 0; i < 3; i++)
200  {
201  BSReference BSRef = BSToFind[i];
202  float BSWeight = LS.LSbsw[i] * 100.0f * BSRef.influence;
203  for (int m = 0; m<BSRef.meshes.Count; m++)
204  {
205  BSRef.meshes[m].SetBlendShapeWeight(BSRef.BSIndices[m], BSWeight);
206  }
207  }

217  wasPlaying = true;

262  // Looks for blend shapes names in the children
263  private void findBlendShapes()
264  {
265  // Find blend shapes in children
266  // References to mesh and blend shape index
267  BSToFind = new BSReference[3];
268  BSToFind[0] = new BSReference(kissBlendShapeNames);
269  BSToFind[1] = new BSReference(lipsClosedBlendShapeNames);
270  BSToFind[2] = new BSReference(mouthOpenBlendShapeNames);
271 
272  // Find children with blend shapes
273  for (int i = 0; i < transform.childCount; i++)
274  {
275  var child = transform.GetChild(i);
276  if (child.GetComponent<SkinnedMeshRenderer>())
277  if (child.GetComponent<SkinnedMeshRenderer>().sharedMesh.blendShapeCount != 0)
278  {
279  var mesh = child.GetComponent<SkinnedMeshRenderer>().sharedMesh;
280  var BSMesh = child.GetComponent<SkinnedMeshRenderer>();
281  // Find blend shape correspondance
282  for (var j = 0; j < mesh.blendShapeCount; j++)
283  {
284  // Store references of the mesh and blend shape index
285  // Iterate over kiss, lipsClosed and mouthOpen
286  for (int k = 0; k < BSToFind.Length; k++)
287  {
288  // Iterate over the blend shape names assigned to each lipsync blend shape
289  for (int ss = 0; ss < BSToFind[k].BSNames.Length; ss++)
290  {
291  // If the blend shape name of the mesh corresponds to the one written in the editor, store the mesh, the BS index and that it was found
292  if (string.Equals(BSToFind[k].BSNames[ss], mesh.GetBlendShapeName(j)))
293  {
294  BSToFind[k].meshes.Add(BSMesh);
295  BSToFind[k].BSIndices.Add(j);
296  BSToFind[k].BSNameIsFound[ss] = true;
297  //Debug.Log("Found! " + mesh.GetBlendShapeName(j) + ", in " + mesh.name + ", mesh count " + BSToFind[k].meshes.Count);
298  }
299  }
300  }
301  }
302  // Debug.Log("Object name: " + child.name +", Blend shape name: " + mesh.GetBlendShapeName(j) + ", Blend shape index: " + j);
303  }
304  }
305 
306  // (Debug) Display the blendshapes not found
307  for (int k = 0; k < BSToFind.Length; k++)
308  {
309  for (int ss = 0; ss < BSToFind[k].BSNames.Length; ss++)
310  {
311  if (!BSToFind[k].BSNameIsFound[ss])
312  Debug.LogError("Blend Shape " + BSToFind[k].BSNames[ss] + " not found in children!");
313  }
314  }
315  }

317  // Fade outs the blend shapes to zero
318  private void fadeOutLips()
319  {
320  // Linearly decrease the value of the blend shapes
321  for (var i = 0; i < 3; i++)
322  {
323  BSReference BSRef = BSToFind[i];
324  for (int m = 0; m < BSRef.meshes.Count; m++)
325  {
326  var indx = BSRef.BSIndices[m];
327  var weight = BSRef.meshes[m].GetBlendShapeWeight(indx) * (timeToFadeOut - Time.time);
328  BSRef.meshes[m].SetBlendShapeWeight(indx, weight);
329  }
330  }
331  if (timeToFadeOut < Time.time)
332  {
333  fadingOut = false;
334  // Set blend shapes to zero
335  for (var i = 0; i < 3; i++){
336  BSReference BSRef = BSToFind[i];
337  for (int m = 0; m < BSRef.meshes.Count; m++){
338  var indx = BSRef.BSIndices[m];
339  BSRef.meshes[m].SetBlendShapeWeight(indx, 0);
340  }
341  }
342  }
