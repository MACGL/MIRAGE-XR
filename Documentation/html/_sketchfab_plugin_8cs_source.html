66  public static void CheckValidity(int deskSizeX = 600, int descSizeY = 175)
67  {
68  if (SkfbLabel == null)
69  {
70  SkfbLabel = new GUIStyle(GUI.skin.label)
71  {
72  richText = true
73  };
74  }
75 
76  if (SkfbTextArea == null)
77  {
78  SkfbTextArea = new GUIStyle(GUI.skin.textArea)
79  {
80  fixedWidth = deskSizeX,
81  fixedHeight = descSizeY
82  };
83  }
84 
85  if (SkfbClickableLabel == null)
86  {
87  SkfbClickableLabel = new GUIStyle(EditorStyles.centeredGreyMiniLabel)
88  {
89  richText = true
90  };
91  }
92  }

94  public static string ClickableTextColor(string text)

99  public static void DisplayVersionPopup()
100  {
101  bool update = EditorUtility.DisplayDialog("Plugin update", "A new version is available \n(you have version " + VERSION + ")\nIt's strongly recommended that you update now. The latest version may include important bug fixes and improvements", "Update", "Skip");
102  if (update)
103  {
104  Application.OpenURL(Urls.latestRelease);
105  }
106  }

108  public static void ShowVersionChecking()

111  }
112 
113  public static void ShowVersionCheckError()
114  {
115  Color current = GUI.color;
116  GUI.color = Color.red;
117  GUILayout.Label("An error occured when looking for the latest exporter version\nYou might be using an old and not fully supported version", EditorStyles.centeredGreyMiniLabel);
118  if (GUILayout.Button("Click here to be redirected to release page"))
119  {
120  Application.OpenURL(Urls.latestRelease);
121  }
122  GUI.color = current;
123  }
124 
125  public static void ShowOutdatedVersionWarning(string latestVersion)
126  {
127  Color current = GUI.color;
128  GUI.color = RED_COLOR;
129  GUILayout.Label("New version " + latestVersion + " available (current version is " + VERSION + ")", EditorStyles.centeredGreyMiniLabel);
130  GUILayout.BeginHorizontal();
131  GUILayout.FlexibleSpace();
132  if (GUILayout.Button("Go to release page", GUILayout.Width(150), GUILayout.Height(25)))
133  {
134  Application.OpenURL(Urls.latestRelease);
135  }
136  GUILayout.FlexibleSpace();
137  GUILayout.EndHorizontal();
138  GUI.color = current;
139  }

141  public static void ShowUpToDate(string latestVersion)
142  {

144  GUILayout.Label("Exporter is up to date (version:" + latestVersion + ")", EditorStyles.centeredGreyMiniLabel);
145 
146  GUILayout.FlexibleSpace();
147  if (GUILayout.Button(ClickableTextColor("Help"), SkfbClickableLabel, GUILayout.Height(20)))
148  {

150  }
151 
152  if (GUILayout.Button(ClickableTextColor("Report an issue"), SkfbClickableLabel, GUILayout.Height(20)))
153  {
154  Application.OpenURL(Urls.reportAnIssue);
155  }
156  GUILayout.EndHorizontal();
157  }
