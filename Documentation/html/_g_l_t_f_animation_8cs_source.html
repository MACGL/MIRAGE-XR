52  public ImportResult Import(GLTFAccessor.ImportResult[] accessors, GLTFNode.ImportResult[] nodes, ImportSettings importSettings) {
53  bool multiRoots = nodes.Where(x => x.IsRoot).Count() > 1;
54 
55  ImportResult result = new ImportResult();
56  result.clip = new AnimationClip();
57  result.clip.name = name;
58 
59  result.clip.legacy = importSettings.useLegacyClips;
60 
61  for (int i = 0; i < channels.Length; i++) {
62  Channel channel = channels[i];
63  if (samplers.Length <= channel.sampler) {
64  Debug.LogWarning($"GLTFUtility: Animation channel points to sampler at index {channel.sampler} which doesn't exist. Skipping animation clip.");
65  continue;
66  }
67  Sampler sampler = samplers[channel.sampler];
68 
69  // Get interpolation mode
70  InterpolationMode interpolationMode = importSettings.interpolationMode;
71  if (interpolationMode == InterpolationMode.ImportFromFile) {
72  interpolationMode = sampler.interpolation;
73  }
74  if (interpolationMode == InterpolationMode.CUBICSPLINE) Debug.LogWarning("Animation interpolation mode CUBICSPLINE not fully supported, result might look different.");
75 
76  string relativePath = "";
77 
78  GLTFNode.ImportResult node = nodes[channel.target.node.Value];
79  while (node != null && !node.IsRoot) {
80  if (string.IsNullOrEmpty(relativePath)) relativePath = node.transform.name;
81  else relativePath = node.transform.name + "/" + relativePath;
82 
83  if (node.parent.HasValue) node = nodes[node.parent.Value];
84  else node = null;
85  }
86 
87  // If file has multiple root nodes, a new parent will be created for them as a final step of the import process. This parent fucks up the curve relative paths.
88  // Add node.transform.name to path if there are multiple roots. This is not the most elegant fix but it works.
89  // See GLTFNodeExtensions.GetRoot
90  if (multiRoots) relativePath = node.transform.name + "/" + relativePath;
91 
92  System.Threading.Thread.CurrentThread.CurrentCulture = System.Globalization.CultureInfo.InvariantCulture;
93  float[] keyframeInput = accessors[sampler.input].ReadFloat().ToArray();
94  switch (channel.target.path) {
95  case "translation":
96  Vector3[] pos = accessors[sampler.output].ReadVec3().ToArray();
97  AnimationCurve posX = new AnimationCurve();
98  AnimationCurve posY = new AnimationCurve();
99  AnimationCurve posZ = new AnimationCurve();
100  for (int k = 0; k < keyframeInput.Length; k++) {
101  posX.AddKey(CreateKeyframe(k, keyframeInput, pos, x => -x.x, interpolationMode));
102  posY.AddKey(CreateKeyframe(k, keyframeInput, pos, x => x.y, interpolationMode));
103  posZ.AddKey(CreateKeyframe(k, keyframeInput, pos, x => x.z, interpolationMode));
104  }
105  result.clip.SetCurve(relativePath, typeof(Transform), "localPosition.x", posX);
106  result.clip.SetCurve(relativePath, typeof(Transform), "localPosition.y", posY);
107  result.clip.SetCurve(relativePath, typeof(Transform), "localPosition.z", posZ);
108  break;
109  case "rotation":
110  Vector4[] rot = accessors[sampler.output].ReadVec4().ToArray();
111  AnimationCurve rotX = new AnimationCurve();
112  AnimationCurve rotY = new AnimationCurve();
113  AnimationCurve rotZ = new AnimationCurve();
114  AnimationCurve rotW = new AnimationCurve();
115  for (int k = 0; k < keyframeInput.Length; k++) {
116  // The Animation window in Unity shows keyframes incorrectly converted to euler. This is only to deceive you. The quaternions underneath work correctly
117  rotX.AddKey(CreateKeyframe(k, keyframeInput, rot, x => x.x, interpolationMode));
118  rotY.AddKey(CreateKeyframe(k, keyframeInput, rot, x => -x.y, interpolationMode));
119  rotZ.AddKey(CreateKeyframe(k, keyframeInput, rot, x => -x.z, interpolationMode));
120  rotW.AddKey(CreateKeyframe(k, keyframeInput, rot, x => x.w, interpolationMode));
121  }
122  result.clip.SetCurve(relativePath, typeof(Transform), "localRotation.x", rotX);
123  result.clip.SetCurve(relativePath, typeof(Transform), "localRotation.y", rotY);
124  result.clip.SetCurve(relativePath, typeof(Transform), "localRotation.z", rotZ);
125  result.clip.SetCurve(relativePath, typeof(Transform), "localRotation.w", rotW);
126  break;
127  case "scale":
128  Vector3[] scale = accessors[sampler.output].ReadVec3().ToArray();
129  AnimationCurve scaleX = new AnimationCurve();
130  AnimationCurve scaleY = new AnimationCurve();
131  AnimationCurve scaleZ = new AnimationCurve();
132  for (int k = 0; k < keyframeInput.Length; k++) {
133  scaleX.AddKey(CreateKeyframe(k, keyframeInput, scale, x => x.x, interpolationMode));
134  scaleY.AddKey(CreateKeyframe(k, keyframeInput, scale, x => x.y, interpolationMode));
135  scaleZ.AddKey(CreateKeyframe(k, keyframeInput, scale, x => x.z, interpolationMode));
136  }
137  result.clip.SetCurve(relativePath, typeof(Transform), "localScale.x", scaleX);
138  result.clip.SetCurve(relativePath, typeof(Transform), "localScale.y", scaleY);
139  result.clip.SetCurve(relativePath, typeof(Transform), "localScale.z", scaleZ);
140  break;
141  case "weights":
142  Debug.LogWarning("GLTFUtility: Morph weights in animation is not supported");
143  break;
144  }
145  }
146  return result;
147  }

149  public static Keyframe CreateKeyframe<T>(int index, float[] timeArray, T[] valueArray, Func<T, float> getValue, InterpolationMode interpolationMode) {
150  float time = timeArray[index];
151  Keyframe keyframe;
152 #pragma warning disable CS0618
153  if (interpolationMode == InterpolationMode.STEP) {
154  keyframe = new Keyframe(time, getValue(valueArray[index]), float.PositiveInfinity, float.PositiveInfinity, 1, 1);
155  } else if (interpolationMode == InterpolationMode.CUBICSPLINE) {
156  // @TODO: Find out what the right math is to calculate the tangent/weight values.
157  float inTangent = getValue(valueArray[index * 3]);
158  float outTangent = getValue(valueArray[(index * 3) + 2]);
159  keyframe = new Keyframe(time, getValue(valueArray[(index * 3) + 1]), inTangent, outTangent, 1, 1);
160  } else { // LINEAR
161  keyframe = new Keyframe(time, getValue(valueArray[index]), 0, 0, 0, 0);
162  }
163 #pragma warning restore CS0618
164  return keyframe;
165  }
166  }

168  public static class GLTFAnimationExtensions {
169  public static GLTFAnimation.ImportResult[] Import(this List<GLTFAnimation> animations, GLTFAccessor.ImportResult[] accessors, GLTFNode.ImportResult[] nodes, ImportSettings importSettings) {
170  if (animations == null) return null;
171 
172  GLTFAnimation.ImportResult[] results = new GLTFAnimation.ImportResult[animations.Count];
173  for (int i = 0; i < results.Length; i++) {
174  results[i] = animations[i].Import(accessors, nodes, importSettings);
175  if (string.IsNullOrEmpty(results[i].clip.name)) results[i].clip.name = "animation" + i;
176  }
177  return results;
178  }
