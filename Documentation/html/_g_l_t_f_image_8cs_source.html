36  public IEnumerator CreateTextureAsync(bool linear, Action<Texture2D> onFinish, Action<float> onProgress = null) {
37  if (!string.IsNullOrEmpty(path)) {
38 #if UNITY_EDITOR
39  // Load textures from asset database if we can
40  Texture2D assetTexture = UnityEditor.AssetDatabase.LoadAssetAtPath(path, typeof(Texture2D)) as Texture2D;
41  if (assetTexture != null) {
42  onFinish(assetTexture);
43  if (onProgress != null) onProgress(1f);
44  yield break;
45  }
46 #endif
47 
48 #if !UNITY_EDITOR && ( UNITY_ANDROID || UNITY_IOS )
49  path = "File://" + path;
50 #endif
51  // TODO: Support linear/sRGB textures
52  using(UnityWebRequest uwr = UnityWebRequestTexture.GetTexture(path, true)) {
53  UnityWebRequestAsyncOperation operation = uwr.SendWebRequest();
54  float progress = 0;
55  while (!operation.isDone) {
56  if (progress != uwr.downloadProgress) {
57  if (onProgress != null) onProgress(uwr.downloadProgress);
58  }
59  yield return null;
60  }
61 
62  if (onProgress != null) onProgress(1f);
63 
64  if (uwr.isNetworkError || uwr.isHttpError) {
65  Debug.LogError("GLTFImage.cs ToTexture2D() ERROR: " + uwr.error);
66  } else {
67  Texture2D tex = DownloadHandlerTexture.GetContent(uwr);
68  tex.name = Path.GetFileNameWithoutExtension(path);
69  onFinish(tex);
70  }
71  uwr.Dispose();
72  }
73  } else {
74  Texture2D tex = new Texture2D(2, 2, TextureFormat.ARGB32, true, linear);
75  if (!tex.LoadImage(bytes)) {
76  Debug.Log("mimeType not supported");
77  yield break;
78  } else onFinish(tex);
79  }

83  public class ImportTask : Importer.ImportTask<ImportResult[]> {
84  public ImportTask(List<GLTFImage> images, string directoryRoot, GLTFBufferView.ImportTask bufferViewTask) : base(bufferViewTask) {
85  task = new Task(() => {
86  // No images
87  if (images == null) return;
88 
89  Result = new ImportResult[images.Count];
90  for (int i = 0; i < images.Count; i++) {
91  string fullUri = directoryRoot + images[i].uri;
92  if (!string.IsNullOrEmpty(images[i].uri)) {
93  if (File.Exists(fullUri)) {
94  // If the file is found at fullUri, read it
95  byte[] bytes = File.ReadAllBytes(fullUri);
96  Result[i] = new ImportResult(bytes, fullUri);
97  } else if (images[i].uri.StartsWith("data:")) {
98  // If the image is embedded, find its Base64 content and save as byte array
99  string content = images[i].uri.Split(',').Last();
100  byte[] imageBytes = Convert.FromBase64String(content);
101  Result[i] = new ImportResult(imageBytes);
102  }
103  } else if (images[i].bufferView.HasValue && !string.IsNullOrEmpty(images[i].mimeType)) {
104  GLTFBufferView.ImportResult view = bufferViewTask.Result[images[i].bufferView.Value];
105  byte[] bytes = new byte[view.byteLength];
106  view.stream.Position = view.byteOffset;
107  view.stream.Read(bytes, 0, view.byteLength);
108  Result[i] = new ImportResult(bytes);
109  } else {
110  Debug.Log("Couldn't find texture at " + fullUri);
111  }
112  }
113  });
114  }
115  }
