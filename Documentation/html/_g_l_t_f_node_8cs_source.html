66  public override IEnumerator OnCoroutine(Action<float> onProgress = null) {
67  // No nodes
68  if (nodes == null) {
69  if (onProgress != null) onProgress.Invoke(1f);
70  IsCompleted = true;
71  yield break;
72  }
73 
74  Result = new ImportResult[nodes.Count];
75 
76  // Initialize transforms
77  for (int i = 0; i < Result.Length; i++) {
78  Result[i] = new GLTFNode.ImportResult();
79  Result[i].transform = new GameObject().transform;
80  Result[i].transform.name = nodes[i].name;
81  }
82  // Set up hierarchy
83  for (int i = 0; i < Result.Length; i++) {
84  if (nodes[i].children != null) {
85  int[] children = nodes[i].children;
86  Result[i].children = children;
87  for (int k = 0; k < children.Length; k++) {
88  int childIndex = children[k];
89  Result[childIndex].parent = i;
90  Result[childIndex].transform.parent = Result[i].transform;
91  }
92  }
93  }
94  // Apply TRS
95  for (int i = 0; i < Result.Length; i++) {
96  nodes[i].ApplyTRS(Result[i].transform);
97  }
98  // Setup components
99  for (int i = 0; i < Result.Length; i++) {
100  // Setup mesh
101  if (nodes[i].mesh.HasValue) {
102  GLTFMesh.ImportResult meshResult = meshTask.Result[nodes[i].mesh.Value];
103  if (meshResult == null) continue;
104 
105  Mesh mesh = meshResult.mesh;
106  Renderer renderer;
107  if (nodes[i].skin.HasValue) {
108  GLTFSkin.ImportResult skin = skinTask.Result[nodes[i].skin.Value];
109  renderer = skin.SetupSkinnedRenderer(Result[i].transform.gameObject, mesh, Result);
110  } else if (mesh.blendShapeCount > 0) {
111  // Blend shapes require skinned mesh renderer
112  SkinnedMeshRenderer mr = Result[i].transform.gameObject.AddComponent<SkinnedMeshRenderer>();
113  mr.sharedMesh = mesh;
114  renderer = mr;
115  } else {
116  MeshRenderer mr = Result[i].transform.gameObject.AddComponent<MeshRenderer>();
117  MeshFilter mf = Result[i].transform.gameObject.AddComponent<MeshFilter>();
118  renderer = mr;
119  mf.sharedMesh = mesh;
120  }
121  //Materials
122  renderer.materials = meshResult.materials;
123  if (string.IsNullOrEmpty(Result[i].transform.name)) Result[i].transform.name = "node" + i;
124  } else {
125  if (string.IsNullOrEmpty(Result[i].transform.name)) Result[i].transform.name = "node" + i;
126  }
127 
128  // Setup camera
129  if (nodes[i].camera.HasValue) {
130  GLTFCamera cameraData = cameras[nodes[i].camera.Value];
131  Camera camera = Result[i].transform.gameObject.AddComponent<Camera>();
132  if (cameraData.type == CameraType.orthographic) {
133  camera.orthographic = true;
134  camera.nearClipPlane = cameraData.orthographic.znear;
135  camera.farClipPlane = cameraData.orthographic.zfar;
136  camera.orthographicSize = cameraData.orthographic.ymag;
137  } else {
138  camera.orthographic = false;
139  camera.nearClipPlane = cameraData.perspective.znear;
140  if (cameraData.perspective.zfar.HasValue) camera.farClipPlane = cameraData.perspective.zfar.Value;
141  if (cameraData.perspective.aspectRatio.HasValue) camera.aspect = cameraData.perspective.aspectRatio.Value;
142  camera.fieldOfView = Mathf.Rad2Deg * cameraData.perspective.yfov;
143  }
144  }
145  }
146  IsCompleted = true;
147  }
148  }

164  private static void CreateNodeListRecursive(Transform transform, List<ExportResult> nodes) {
165  ExportResult node = new ExportResult();
166  node.name = transform.name;
167  node.translation = transform.localPosition;
168  node.rotation = transform.localRotation;
169  node.scale = transform.localScale;
170  node.renderer = transform.gameObject.GetComponent<MeshRenderer>();
171  node.filter = transform.gameObject.GetComponent<MeshFilter>();
172  node.skinnedRenderer = transform.gameObject.GetComponent<SkinnedMeshRenderer>();
173  nodes.Add(node);
174  if (transform.childCount > 0) {
175  if (transform.childCount > 0) {
176  node.children = new int[transform.childCount];
177  for (int i = 0; i < node.children.Length; i++) {
178  Transform child = transform.GetChild(i);
179  node.children[i] = nodes.Count;
180  CreateNodeListRecursive(child, nodes);
181  }
182  }
183  }
