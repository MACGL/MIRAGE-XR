146  public static async Task<(bool, Texture2D)> LoadTextureAsync(string url)
147  {
148  const int timeout = 60 * 2;
149 
150  const string httpPrefix = "http";
151  const string filePrefix = "file://";
152 
153  byte[] bytes;
154  var texture = new Texture2D(2, 2);
155  if (url.StartsWith(httpPrefix))
156  {
157  (_, bytes) = await DownloadBytesAsync(url, timeout);
158  }
159  else if (url.StartsWith(filePrefix))
160  {
161  (_, bytes) = await ReaFileAsync(url.Substring(filePrefix.Length));
162  }
163  else
164  {
165  return (false, null);
166  }
167 
168  var result = bytes != null && texture.LoadImage(bytes);
169  return (result, result ? texture : null);
170  }

209  public static async Task<bool> DownloadModelAndExtractAsync(string url, ModelPreviewItem modelPreview, Action<float> onProgressChanged = null)
210  {
211  const int timeout = 5 * 60;
212 
213  var archiveName = $"{modelPreview.name}.zip";
214  var imageName = $"{modelPreview.name}.png";

219  if (!Directory.Exists(modelsFolderPath)) Directory.CreateDirectory(modelsFolderPath);
220 
221  var modelFolderPath = Path.Combine(modelsFolderPath, modelPreview.name);
222  if (!Directory.Exists(modelFolderPath)) Directory.CreateDirectory(modelFolderPath);
223 
224  var archivePath = Path.Combine(modelsFolderPath, archiveName);
225  var jsonPath = Path.Combine(modelFolderPath, JSON_FILE_NAME);

228 
229  using (var stream = new FileStream(archivePath, FileMode.OpenOrCreate))
230  {

237  }
238 
239  if (!Directory.Exists(modelFolderPath)) Directory.CreateDirectory(modelFolderPath);
240 
241  using (var stream = new FileStream(imagePath, FileMode.OpenOrCreate))
242  {
243  var (result, _) = await Network.DownloadToStreamAsync(modelPreview.resourceImage.url, stream, timeout);
244  if (!result)

248  modelPreview.resourceImage.url = $"file://{imagePath}";
