249  IEnumerator RequestToken(string url, string code, Action<string> callbackTokenReceived)
250  {
251  string redirectUri = String.Empty;
252 #if UNITY_EDITOR
253  redirectUri = LoopbackURLlocal;
254 #elif WINDOWS_UWP
255  redirectUri = LoopbackURL;
256 #endif
257 
258  WWWForm form = new WWWForm();
259  form.AddField("client_id", appID);
260  form.AddField("redirect_uri", redirectUri);
261  form.AddField("code", code);
262  if (responseType == "code")
263  form.AddField("grant_type", "authorization_code");
264  if (appSecret != "")
265  form.AddField("client_secret", appSecret);
266  foreach (KeyValuePair<string, string> field in form.headers)
267  Debug.Log($"{field.Key}: {field.Value}");
268 
269  UnityWebRequest www = UnityWebRequest.Post(url, form);
270  yield return www.SendWebRequest();
271  string response;
272  if (www.error != null)
273  {
274  onAuthenticationFailedOrCancelled?.Invoke($"{www.error}: {www.downloadHandler.text}");
275  callbackTokenReceived(null);
276  }
277  else
278  {
279  response = (www.downloadHandler.text);
280  try
281  {
282  AccessTokenResponseJson tokenJson = JsonUtility.FromJson<AccessTokenResponseJson>(response);
283  if (!String.IsNullOrEmpty(tokenJson.access_token))
284  {
285  isAuthenticated = true;
286  onTokenReceivedSuccessfully?.Invoke();
287  callbackTokenReceived(tokenJson.access_token);
288  }
289  else
290  {
291  onAuthenticationFailedOrCancelled?.Invoke("Access token is empty");
292  callbackTokenReceived(null);
293 
294  }
295 
296  }
297  catch (Exception e)
298  {
299  Debug.Log(e.Message + "," + e.StackTrace);
300  onAuthenticationFailedOrCancelled?.Invoke($"{e.Message},{e.StackTrace}");
301  }
302  }
303  yield return null;
