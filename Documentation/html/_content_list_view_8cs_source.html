280  private static IEnumerator SizeTo(RectTransform rectTransform, RectTransform.Axis axis, float sizeEnd, float time, AnimationCurve curve = null, Action callback = null) {
281  if (curve == null) curve = AnimationCurve.Linear(0f, 0f, 1f, 1f);
282  var rect = rectTransform.rect;
283  var sizeStart = axis == RectTransform.Axis.Horizontal ? rect.width : rect.height;
284  var timer = 0.0f;
285  while (timer < 1.0f) {
286  timer = Mathf.Min(1.0f, timer + Time.deltaTime/time);
287  var value = curve.Evaluate(timer);
288  var size = Mathf.Lerp(sizeStart, sizeEnd, value);
289  rectTransform.SetSizeWithCurrentAnchors(axis, size);
290 
291  yield return null;
292  }
293 
294  callback?.Invoke();
295  }
