297  private static async Task<byte[]> CompressRecord(string path, string recordingId)
298  {
299  byte[] bytes = null;
300  try
301  {
302  using (var stream = new MemoryStream())
303  {
304  using (var zipStream = new ZipOutputStream(stream))
305  {
306  if (Directory.Exists(path))
307  await ZipUtilities.CompressFolderAsync(path, zipStream);
308  await ZipUtilities.AddFileToZipStreamAsync(zipStream,
309  $"{path}-activity.json", $"{recordingId}-activity.json");
310  await ZipUtilities.AddFileToZipStreamAsync(zipStream,
311  $"{path}-workplace.json", $"{recordingId}-workplace.json");
312  }
313  bytes = stream.ToArray();
314  }
315  }
316  catch (Exception e)
317  {
318  Debug.LogError($"compression error: {e}");
319  }
320  return bytes;

323  public static async Task<(bool, Activity)> DownloadActivity(Session session)
324  {
325  var isTooBigForMemory = session.filesize > MAX_FILE_SIZE_FOR_MEMORY;
326 
327  bool result;
328  Activity activity = null;
329  var tempFilePath = Path.Combine(Application.persistentDataPath, Guid.NewGuid().ToString());
330  var stream = isTooBigForMemory ? File.OpenWrite(tempFilePath) : (Stream)new MemoryStream();
331  try
332  {
333  string error;
334  (result, error) = await Network.DownloadRequestToStreamAsync(stream, session.contextid, session.component, session.filearea, session.itemid, session.filename, DBManager.domain);
335  if (result)
336  {
337  if (isTooBigForMemory)
338  {
339  stream.Dispose();
340  stream = File.OpenRead(tempFilePath);
341  }
342  await ZipUtilities.ExtractZipFileAsync(stream, Application.persistentDataPath);
343  var fileName = LocalFiles.GetActivityJsonFilename(session.sessionid, true);
344  var newActivity = await LocalFiles.ReadActivityAsync(fileName);
345  if (newActivity != null) activity = newActivity;
346  }
347  else
348  {
349  Debug.LogError(error);
350  }
351  }
352  catch (Exception e)
353  {
354  Debug.LogError(e);
355  result = false;
356  }
357  finally
358  {
359  stream.Dispose();
360  if (isTooBigForMemory && File.Exists(tempFilePath)) File.Delete(tempFilePath);
361  }
362 
363  return (result, activity);
364  }
