287  private IEnumerator RequestToken(string code, Action<string> callbackTokenReceived)
288  {
289  const string redirectUri = "https://wekit-community.org/sketchfab/callback.php";
290  const string postUri = "https://sketchfab.com/oauth2/token/";
291 
292  var appID = SketchfabDataObject.clientData.ClientId;
293  var appSecret = SketchfabDataObject.clientData.ClientSecret;
294 
295  var form = new WWWForm();
296  form.AddField("client_id", appID);
297  form.AddField("redirect_uri", redirectUri);
298  form.AddField("code", code);
299  form.AddField("grant_type", "authorization_code");
300  form.AddField("client_secret", appSecret);
301 
302  var www = UnityWebRequest.Post(postUri, form);
303  www.SetRequestHeader("Content-Type", "application/x-www-form-urlencoded");
304 
305  yield return www.SendWebRequest();
306  if (www.error != null)
307  {
308  callbackTokenReceived("error");
309  }
310  else
311  {
312  var response = (www.downloadHandler.text);
313  var tokenJson = JsonUtility.FromJson<AccessTokenResponseJson>(response);
314 
315  callbackTokenReceived(!string.IsNullOrEmpty(tokenJson.access_token) ? tokenJson.access_token : "empty");
316  }
317  yield return null;
318  }

433  public static IEnumerator GetSketchfabResults(string url, string token, Action<string> callbackReceivedCode)
434  {
435  const float cooldown = 0.5f;
436  var retryCount = 0;
437  while (retryCount < RETRY_MAX_COUNT)
438  {
439  retryCount++;
440 
441  var request = new UnityWebRequest(url);
442  request.SetRequestHeader("Authorisation", $"Bearer {token}");
443  var handler = new DownloadHandlerBuffer();
444  request.downloadHandler = handler;
445 
446  yield return request.SendWebRequest();
447 
448  if (!request.isNetworkError && !request.isHttpError)
449  {
450  if (request.responseCode == 200)
451  {
452  Debug.Log("Request finished successfully");
453  callbackReceivedCode(handler.text);
454  break;
455  }
456 
457  Debug.Log(request.responseCode == 401
458  ? "Error 401: Unauthorized. Resubmitted request!"
459  : $"Request failed (status:{request.responseCode})");
460  }
461 
462  yield return new WaitForSeconds(cooldown);
463  }
464  }

466  public static IEnumerator DownloadImage(string url, Action<Sprite> callback)
467  {
468  const string httpPrefix = "http";
469  const string filePrefix = "file://";
470 
471  var texture = new Texture2D(2, 2);
472  if (url.StartsWith(httpPrefix))
473  {
474  var www = UnityWebRequestTexture.GetTexture(url);
475  yield return www.SendWebRequest();
476 
477  if (!www.isNetworkError && !www.isHttpError)
478  {
479  texture = ((DownloadHandlerTexture) www.downloadHandler).texture;
480  }
481  }
482  else if (url.StartsWith(filePrefix))
483  {
484  var imagePath = url.Substring(filePrefix.Length);
485  if (File.Exists(imagePath))
486  {
487  var fileData = File.ReadAllBytes(imagePath);
488  texture.LoadImage(fileData);
489  }
490  }
491  else
492  {
493  Debug.Log("Unrecognised link for thumbnail image");
494  }
495 
496  var rec = new Rect(0, 0, texture.width, texture.height);
497  var ppu = Mathf.Max(texture.width / 9f, texture.height / 6f);
498  var sprite = Sprite.Create(texture, rec, new Vector2(0.5f, 0.5f), ppu);
499 
500  callback?.Invoke(sprite);
501  }
