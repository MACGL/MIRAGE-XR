433  public static async Task CopyToAsync(this Stream source, Stream destination, int bufferSize, IProgress<long> progress = null, CancellationToken cancellationToken = default)
434  {
435  if (source == null)
436  {
437  throw new ArgumentNullException(nameof(source));
438  }
439 
440  if (!source.CanRead)
441  {
442  throw new ArgumentException("Has to be readable", nameof(source));
443  }
444 
445  if (destination == null)
446  {
447  throw new ArgumentNullException(nameof(destination));
448  }
449 
450  if (!destination.CanWrite)
451  {
452  throw new ArgumentException("Has to be writable", nameof(destination));
453  }
454 
455  if (bufferSize < 0)
456  {
457  throw new ArgumentOutOfRangeException(nameof(bufferSize));
458  }
459 
460  var buffer = new byte[bufferSize];
461  long totalBytesRead = 0;
462  int bytesRead;
463  while ((bytesRead = await source.ReadAsync(buffer, 0, buffer.Length, cancellationToken).ConfigureAwait(false)) != 0)
464  {
465  await destination.WriteAsync(buffer, 0, bytesRead, cancellationToken).ConfigureAwait(false);
466  totalBytesRead += bytesRead;
467  progress?.Report(totalBytesRead);
468  }
469  }
