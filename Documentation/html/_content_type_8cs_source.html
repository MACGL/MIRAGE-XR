175  public static ContentType ParsePredicate(string predicate)
176  {
177  var predicateLowCase = predicate.ToLower();
178  if (predicateLowCase.Contains(PREDICATE_IMAGE))
179  return ContentType.IMAGE;
180  if (predicateLowCase.Contains(PREDICATE_VIDEO))
181  return ContentType.VIDEO;
182  if (predicateLowCase.Contains(PREDICATE_AUDIO) || predicateLowCase.Contains(PREDICATE_AUDIO_V2))
183  return ContentType.AUDIO;
184  if (predicateLowCase.Contains(PREDICATE_GHOST) || predicateLowCase.Contains(PREDICATE_GHOST_V2))
185  return ContentType.GHOST;
186  if (predicateLowCase.Contains(PREDICATE_LABEL))
187  return ContentType.LABEL;
188  if (predicateLowCase.Contains(PREDICATE_ACT))
189  return ContentType.ACT;
190  if (predicateLowCase.Contains(PREDICATE_VFX))
191  return ContentType.VFX;
192  if (predicateLowCase.Contains(PREDICATE_MODEL) || predicateLowCase.Contains(PREDICATE_MODEL_V2))
193  return ContentType.MODEL;
194  if (predicateLowCase.Contains(PREDICATE_CHARACTER))
195  return ContentType.CHARACTER;
196  if (predicateLowCase.Contains(PREDICATE_PICKANDPLACE) || predicateLowCase.Contains(PREDICATE_PICKANDPLACE_V2))
197  return ContentType.PICKANDPLACE;
198  if (predicateLowCase.Contains(PREDICATE_IMAGEMARKER))
199  return ContentType.IMAGEMARKER;
200  if (predicateLowCase.Contains(PREDICATE_PLUGIN))
201  return ContentType.PLUGIN;
202 
203  return ContentType.UNKNOWN;
204  }
