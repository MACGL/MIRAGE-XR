58  void GenerateBoundaryUVs(Mesh mesh)
59  {
60  int vertexCount = mesh.vertexCount;
61 
62  // Reuse the list of UVs
63  s_FeatheringUVs.Clear();
64  if (s_FeatheringUVs.Capacity < vertexCount) { s_FeatheringUVs.Capacity = vertexCount; }
65 
66  mesh.GetVertices(s_Vertices);
67 
68  Vector3 centerInPlaneSpace = s_Vertices[s_Vertices.Count - 1];
69  Vector3 uv = new Vector3(0, 0, 0);
70  float shortestUVMapping = float.MaxValue;
71 
72  // Assume the last vertex is the center vertex.
73  for (int i = 0; i < vertexCount - 1; i++)
74  {
75  float vertexDist = Vector3.Distance(s_Vertices[i], centerInPlaneSpace);
76 
77  // Remap the UV so that a UV of "1" marks the feathering boudary.
78  // The ratio of featherBoundaryDistance/edgeDistance is the same as featherUV/edgeUV.
79  // Rearrange to get the edge UV.
80  float uvMapping = vertexDist / Mathf.Max(vertexDist - FeatheringWidth, 0.001f);
81  uv.x = uvMapping;
82 
83  // All the UV mappings will be different. In the shader we need to know the UV value we need to fade out by.
84  // Choose the shortest UV to guarentee we fade out before the border.
85  // This means the feathering widths will be slightly different, we again rely on a fairly uniform plane.
86  if (shortestUVMapping > uvMapping) { shortestUVMapping = uvMapping; }
87 
88  s_FeatheringUVs.Add(uv);
89  }
90 
91  m_FeatheredPlaneMaterial.SetFloat("_ShortestUVMapping", shortestUVMapping);
92 
93  // Add the center vertex UV
94  uv.Set(0, 0, 0);
95  s_FeatheringUVs.Add(uv);
96 
97  mesh.SetUVs(1, s_FeatheringUVs);
98  mesh.UploadMeshData(false);
99  }
